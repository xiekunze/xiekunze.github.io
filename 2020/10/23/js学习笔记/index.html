3.let&const暂时性死区TDC (规范使用)
let 变量只能先声明再使用
例:

<script>
        document.write(i); //Uncaught ReferenceError: Cannot access 'i' before initialization
        let i = 4;
</script>

const与let 一致(两个都是全局变量)
const与let的不同

<script>
        let i = 4;
        function dd(){
            i = 5;
            document.write(i+"<br>");
        }
        dd(); //5
        document.write(i); //5
</script>

<script>
        const i = 4;
        function dd(){
            i = 5;
            document.write(i+"<br>");  //报错
        }
        dd();
        document.write(i);  //报错
        //Uncaught TypeError: Assignment to constant variable.
        //at dd (new_file.html:11)
        //at new_file.html:14
</script>

函数内有变量就使用自己的,如果变量不声明(没有)就去上一层找!

9.关于值与地址

    <script>
        var i = 10;
        var b = i;
        document.write(i,b); //10 10
        b = 11;  //如果是值的话, 会另外开辟一块空间,改变了谁，谁的值就变！
        document.write(i,b); //10 11

        let p = {name: "xiekunze"}; //一般是对象,数组(引用类型)的话,会共用一块内存空间！

        document.write(l.name); //ddddd
        document.write(p.name); //ddddd
    </script>

使用规范

作用域链只向上查找，找到全局window即终止，应该尽量不要在全局作用域中添加变量。 (后盾人的例子)

函数被执行后其环境变量将从内存中删除。下面函数在每次执行后将删除函数内部的a变量。
这破玩意儿,调用一次,申请一次内存,每次调用内存地址都不一样,调用完以后就释放内存!

<script>
        function aa(){
            let a = 0;
            document.write(a);  //0
        }
        aa();
</script>

函数每次调用都会创建一个新作用域

<script>
        let a = 11;
        function aa(){
            let b = 22;
            console.log(a);   //11
            console.log(b);      //22
            function bb(){
                let c = 33;
                console.log(b);   //22
                console.log(c);      //33
            }
            bb();
        }
        aa();
</script>

函数作用域中可以用全局作用域的东西(如:变量),全局作用域中的东西不能用函数作用域中东西(如:变量)
当函数中嵌套函数,即嵌套的函数相当于全局函数,里面的东西(如:变量)可以直接被嵌套的函数所使用!

(延伸函数环境生命周期)
如果子函数被使用时父级环境将被保留(用return的话,会保留,但具体哪一层要用()去决定)
环境如果被使用(被赋值就不删除)

<script>
        function hd() {
          let n = 1;
          return function() {
            let b = 1;
            console.log(5);  //5
            return function() {
              console.log(++n);
              console.log(++b);
            };
          };
        }
        let a = hd()();//5  对应第一层
        a(); //2,2  对应第二层
        a(); //3,3  再次调用第二层 (数会类计下下来)
</script>

只要是你没有用return,用调用函数的话,即失去了父级环境!

<script>
        function aa(){
            let a = 1;
            return function bb(){
                let b = 2;
                function cc(){
                    console.log(++b);
                }
                cc();
            }
        }
        let f = aa();
        f();   //3
        f();   //3
</script>

构造函数也是很好的环境例子，子函数被外部使用父级环境将被保留

<script>
        function User() {
          let a = 1;
          this.show = function() {
            console.log(a++);
          };
        }
        let a = new User();
        a.show(); //1
        a.show(); //2
        let b = new User();
        b.show(); //1
</script>

在 for 循环中使用var 会在每一次迭代中重新生成同个变量(因为使用的是window)
不管你需要哪个,都会走完一次流程再返回结果!

<script>

        for(var i = 0; i < 3 ; i++){
            arr.push(function(){
                return i; 
            })
        }
        console.log(arr[1]());  //3
</script>

<script>

                    return i;

抽象特性 (后盾人的例子)

即里面的不让外面的操作!

将复杂功能隐藏在内部，只开放给外部少量方法，更改对象内部的复杂逻辑不会对外部调用造成影响即抽象。

问题分析

代理拦截

对 函数,对象,数组 进行操作!

代理（拦截器）是对象的访问控制，setter/getter 是对单个对象属性的控制，而代理是对整个 对象 的控制。

  • 读写属性时代码更简洁

  • 对象的多个属性控制统一交给代理完成

  • 严格模式下 set 必须返回布尔值

不直接new对象,通过 new Proxy 来代理整个对象!

<script>
        "use strict";
        const hd = { name: "后盾人" };
        const proxy = new Proxy(hd, {
          get(obj, property) {
            return obj[property];  //严格模式下有return要求
          },
          set(obj, property, value) {
            obj[property] = value;
            return true;  //严格模式下有return要求
          }
        });
        proxy.age = 10;
        console.log(hd);//{"age":10,"name":"后盾人"}
</script>

代理函数

所以所有的操作都在Proxy()里面操作!

如果代理以函数方式执行时，会执行代理中定义 apply 方法。

  • 参数说明：函数，上下文对象，参数

下面使用 apply 计算 函数 执行时间

<script>
        "use strict";
        function factorial(num) {
          return num == 1 ? 1 : num * factorial(num - 1);
        }
        let proxy = new Proxy(factorial, {  //函数作为参数
          apply(func, obj, args) {
            console.time("run");
            func.apply(obj, args);
            console.timeEnd("run");
          }
        });
        proxy.apply(this, [1, 2, 3]);
</script>

双向绑定(后盾人的例子)

下面通过代理实现vue 等前端框架的数据绑定特性特性。

<script>
        "use strict";
        function View() {
            //设置代理拦截
          let proxy = new Proxy(
            {},
            {
              get(obj, property) {},
              set(obj, property, value) {
                obj[property] = value;
                document
                  .querySelectorAll(
                    `[v-model="${property}"],[v-bind="${property}"]`
                  )
                  .forEach(el => {
                    el.innerHTML = value;
                    el.value = value;
                  });
              }
            }
          );
          //初始化绑定元素事件
          this.run = function() {
            const els = document.querySelectorAll("[v-model]");
            els.forEach(item => {
              item.addEventListener("keyup", function() {
                proxy[this.getAttribute("v-model")] = this.value;
              });
            });
          };
        }
        let view = new View().run();
</script>
